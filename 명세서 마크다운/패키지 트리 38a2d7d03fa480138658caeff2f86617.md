```python
├── alembic/                                      # Alembic DB 마이그레이션
│
├── app/
│
│   ├── core/                                     # 프로젝트 공통 설정

│   │   ├── logger.py                             # 로그 설정
│   │   ├── constants.py                          # 프로젝트 상수

│   │
│   ├── db/                                       # 영속(DB) 데이터
│   │   ├── database.py                           # DB 연결
│   │   │
│   │   ├── models/                               # SQLAlchemy Model
│   │   │
│   │   ├── schema/                               # Pydantic Schema
│   │   │
│   │   └── crud/                                 # DB CRUD
│   │       ├── menuCrud.py                       # 메뉴 CRUD
│   │       ├── optionCrud.py                     # 옵션 CRUD
│   │       ├── orderCrud.py                      # 주문 CRUD
│   │       ├── paymentCrud.py                    # 결제 CRUD
│   │       └── logCrud.py                        # 주문 로그 CRUD
│   │
│   ├── memory/                                   # 메모리(Session)

│   │   ├── session/                                  # Session 도메인 객체
│   │   │   ├── session.py                            # Session
│   │   │   ├── orderItem.py                          # 현재 주문중인 메뉴
│   │   │   ├── cartItem.py                           # 장바구니 Item
│   │   │   └── enums.py                              # FSMState, OrderItemStatus
│   │   │   ├── sessionMemory.py                      # Session 저장(Map, SessionID 생성)
│   │   │   └── sessionCrud.py                        # Session 메모리 CRUD
│   │
│   ├── ai/                                       # 음성 처리
│   │
│   │   ├── pipeline/
│   │   │   ├── voicePipeline.py                  # STT → Rule → AI → FSM 전체 흐름

│   │   │   └── pipelineContext.py                # Pipeline 공유 데이터
│   │   │
│   │   ├── stt/

│   │   │
│   │   ├── rapidfuzz/
│   │   │   └── normalizer.py                     # 메뉴명 보정
│   │   │
│   │   ├── ruleEngine/
│   │   │   ├── ruleParser.py                     # NormalizeResult → ParseResult
│   │   │   ├── ruleEngine.py                     # ParseResult → EventList
│   │   │   ├── rules.py                          # Rule 정의
│   │   │   └── eventExecutor.py                  # EventList 실행
│   │   │
│   │   └── client/
│   │       ├── aiClient.py                       # AI Server 호출(Client)
│   │       ├── request.py                        # 요청 생성 
│   │       └── response.py                       # 응답 변환  
│   │
│   ├── fsm/                                      # FSM
│   │   ├── state.py                              # FSM 상태 정의
│   │   ├── event.py                              # Event 정의
│   │   ├── transition.py                         # 상태 전이
│   │   ├── validator.py                          # Event 검증
│   │   └── dispatcher.py                         # FSM.dispatch()
│   │
│   ├── controllers/                              # 비즈니스 로직
│   │   ├── orderController.py                    # 주문 처리
│   │   ├── cartController.py                     # 장바구니 처리
│   │   ├── recommendationController.py           # 추천 처리
│   │   ├── paymentController.py                  # 결제 처리
│   │   └── systemController.py                   # 시스템 처리
│   │
│   ├── interface/                                # 외부 인터페이스
│   │
│   │   ├── websocket/
│   │   │   ├── manager.py                        # WebSocket 연결 관리
│   │   │   ├── handler.py                        # WebSocket 송수신
│   │   │   └── message.py                        # Message Type
│   │   │
│   │   └── dto/
│   │       ├── voiceRequest.py                   # Android 음성 요청
│   │       ├── sttResult.py                      # STT 결과
│   │       ├── normalizeResult.py                # RapidFuzz 결과
│   │       ├── parseResult.py                    # RuleParser 결과
│   │       ├── eventList.py                      # RuleEngine 결과
│   │       ├── llmResult.py                      # AI Server 응답(Event, Response)
│   │       ├── sessionResponse.py                # Android 응답
│   │       ├── paymentResultRequest.py           # 결제 결과 요청
│   │       └── errorResponse.py                  # 오류 응답
│   │
│   ├── routers/                                  # FastAPI Router
│   │   ├── websocketRouter.py                    # 음성(WebSocket)
│   │   ├── sessionRouter.py                      # Session 생성
│   │   └── paymentRouter.py                      # 결제 완료
│   │
│   ├── middleware/                               # FastAPI Middleware
│   │
│   ├── utils/                                    # 공통 유틸
│   │
│   └── main.py                                   # FastAPI 실행
│
├── requirements.txt
│
└── .env
```
