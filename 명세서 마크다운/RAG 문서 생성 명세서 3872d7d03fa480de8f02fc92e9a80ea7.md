CSV → DB 저장된 값을 레그문서로 만드는 함수

```jsx
def build_menu_document(menu_data):

    lines = []

    lines.append(f"메뉴명: {menu_data['menu_name']}")
    lines.append("")
    lines.append(f"카테고리: {menu_data['category']}")
    lines.append("")

    lines.append("설명:")
    lines.append(menu_data["description"] or "")
    lines.append("")

    lines.append("가격:")
    lines.append(f"{menu_data['price']}원")
    lines.append("")

    lines.append("원재료:")
    lines.append(
        ", ".join(menu_data["ingredients"])
        if menu_data["ingredients"]
        else "없음"
    )
    lines.append("")

    lines.append("알러지:")
    lines.append(
        ", ".join(menu_data["allergies"])
        if menu_data["allergies"]
        else "없음"
    )
    lines.append("")

    lines.append("선택 가능 옵션:")
    lines.append("")

    for group in menu_data["option_groups"]:

        lines.append(f"[{group['name']}]")
        lines.append("")

        lines.append(
            f"필수여부: {'필수' if group['required'] else '선택'}"
        )

        lines.append(
            f"최소선택: {group['min']}"
        )

        lines.append(
            f"최대선택: {group['max']}"
        )

        lines.append("")

        for option in group["options"]:

            lines.append(
                f"- {option['name']} (+{option['price']}원)"
            )

        lines.append("")

    return "\n".join(lines)
```

메타데이터 생성용 코드

```jsx
def get_menu_data(db, menu_id):

    menu = (
        db.query(Menu)
        .filter(Menu.m_id == menu_id)
        .first()
    )

    menu_data = {
        "menu_id": menu.m_id,
        "menu_name": menu.m_name,
        "category": menu.category.c_name,
        "description": menu.m_description,
        "price": menu.m_price,

        "ingredients": [],
        "allergies": [],
        "option_groups": []
    }

    for ingredient in menu.ingredients:
        menu_data["ingredients"].append(
            ingredient.i_name
        )

    for allergy in menu.allergies:
        menu_data["allergies"].append(
            allergy.a_name
        )

    for group in menu.option_groups:

        group_data = {
            "name": group.og_name,
            "required": group.og_required,
            "min": group.og_min,
            "max": group.og_max,
            "options": []
        }

        for option in group.options:

            group_data["options"].append(
                {
                    "name": option.op_name,
                    "price": option.op_price
                }
            )

        menu_data["option_groups"].append(
            group_data
        )

    return menu_data
```

초기 실행용(데이터 적재용)

```jsx
menus = db.query(Menu).all()

for menu in menus:

    menu_data = get_menu_data(
        db,
        menu.m_id
    )

    document = build_menu_document(
        menu_data
    )

    collection.add(
        ids=[f"menu_{menu.m_id}"],
        documents=[document],
        metadatas=[
            {
                "menu_id": menu.m_id,
                "menu_name": menu.m_name
            }
        ]
    )
```
